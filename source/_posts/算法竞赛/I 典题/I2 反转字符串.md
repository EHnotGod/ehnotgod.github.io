### Python代码实现

```python
import sys
input = sys.stdin.readline

t = int(input())
for _ in range(t):
    s = list(map(int, input().strip()))
    n = len(s); new = []
    for i in range(n):
        if s[i] != 2:
            new.append(((i + 1) % 2) ^ s[i])
        else:
            new.append(s[i])
    c0 = new.count(0); c1 = new.count(1); c2 = new.count(2)
    A = abs(c0 - c1)
    if c2 < A:
        ans = A - c2
    else:
        ans = (c2 - A) % 2
    print(ans)
```
